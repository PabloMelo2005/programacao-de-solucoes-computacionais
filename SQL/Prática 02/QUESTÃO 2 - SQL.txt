CREATE TABLE CLIENTE (
  CPF varchar(11) PRIMARY KEY, 
  NOME varchar(100), 
  DATA_NASCIMENTO date, 
  TELEFONE varchar(15), 
  ENDEREÇO varchar(100)
);

CREATE TABLE VEÍCULO (
  CHASSI varchar(17) PRIMARY KEY,
  PLACA varchar(7),
  MODELO varchar(30),
  MARCA varchar(10),
  ANO varchar(4)
  );

CREATE TABLE MECÂNICO (
 CPF varchar(11), 
 NOME varchar(100), 
 DATA_NASCIMENTO date, 
 TELEFONE varchar(15), 
 ENDEREÇO varchar(100), 
 SALARIO decimal(10,2)
  );
