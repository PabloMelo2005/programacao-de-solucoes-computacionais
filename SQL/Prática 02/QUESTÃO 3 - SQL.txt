CREATE TABLE CLIENTE (
  CPF varchar(11) PRIMARY KEY, 
  NOME varchar(100), 
  DATA_NASCIMENTO date, 
  AGENCIA varchar(20), 
  CONTA varchar(20), 
  TELEFONE varchar(15),
  ENDEREÇO varchar(20),
  PERFIL varchar(20) CHECK (PERFIL IN ('conservador', 'moderado', 'arrojado', 'sofisticado', 'agressivo'))
  );

CREATE TABLE ATIVO_FINANCEIRO (
  CÓDIGO int PRIMARY KEY, 
  NOME varchar(20), 
  DESCRIÇÃO text,
  EMISSOR varchar(100),
  DATA_INÍCIO date,
  DATA_FIM date 
);

CREATE TABLE NOTA_CORRETAGEM (
  ID INTEGER PRIMARY KEY AUTOINCREMENT,
  CPF_CLIENTE VARCHAR(11),
  DATA_COMPRA DATE, 
  CODIGO_ATIVO INT,
  QTDE INT, 
  COTACAO DECIMAL(10,2),
  VALOR DECIMAL(10,2),
  FOREIGN KEY (CPF_CLIENTE) REFERENCES CLIENTE(CPF),
  FOREIGN KEY (CODIGO_ATIVO) REFERENCES ATIVO_FINANCEIRO(CÓDIGO)
);
